Chia quà - ntucoder

2020-10-30
Quy hoạch động ntucoder , Bignum

Link đề bài:

  • http://ntucoder.net/Problem/Details/4414.

Hướng dẫn:

  • Gọi f[i][j] là số cách chia các món quà từ 1 tới i món quà, vào j túi.
  • Nếu chia món quà thứ i vào một túi mới thì f[i][j] = f[i-1][j-1]
  • Nếu bỏ món quà đó vào các túi sẵn có f[i][j] = f[i-1][j] * j

Code mẫu:

#include <bits/stdc++.h>
#define JN Jonasngyn
#define pb push_back
#define FOR(i, a, b) for(int i = int(a); i <= int(b); i++)
#define FORD(i, a, b) for(int i = int(a); i >= int(b); i--)
#define REP(i, r) for(int i = 0; i < r; i++)
#define FAST ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
#define READFILE freopen("input.txt", "r", stdin)
#define WRITEFILE freopen("output.txt", "w", stdout);
#define ll long long
#define ii pair<int, int>
#define fi first
#define se second
using namespace std;
typedef vector<ll> bigNum;
const int BASE=1e9;
const int MAX=111;

int n,m;

bigNum toBig(int x){
	bigNum a;
	a.push_back(x);
	return a;
}

bigNum Fix(bigNum &a)
{
    a.push_back(0);
    for(int i=0; i<(int)(a.size()-1); i++)
    {
        a[i+1]+=a[i]/BASE;
        a[i]%=BASE;
        while (a[i]<0) a[i+1]--, a[i]+=BASE;
    }
    while (a.size()>=2 && a.back()==0) a.pop_back();
    return a;
}

bigNum operator +(bigNum a, bigNum b)
{
	a.resize(max(a.size(),b.size()));
	for(int i=0; i<(int)b.size(); i++) a[i]+=b[i];
	return Fix(a);
}

bigNum operator * ( bigNum a, bigNum b )
{
    bigNum ret(a.size()+b.size()+2);
    REP(i,a.size())
        REP(j,b.size())
        {
            ret[i+j]+=a[i]*b[j];
            ret[i+j+1]+=(ret[i+j])/BASE;
            ret[i+j]%=BASE;
        }
    return Fix(ret);
}

bigNum ss(bigNum a, bigNum b)
{
	a.resize(max(a.size(),b.size()));
	b.resize(max(a.size(),b.size()));
	FORD(i,a.size()-1,0)
	{
		if (a[i]>b[i]) return a;
		if (a[i]<b[i]) return b; 
	}
	return a;
}

void Print(bigNum a)
{
	printf("%d", a.back());
	for(int i=a.size()-2; i>=0; i--)
	{
		printf("%09d", a[i]);
	}
}

bigNum f[MAX][MAX];

int main()
{
    #ifndef ONLINE_JUDGE
        READFILE;
    #endif
    FAST;
    
 	cin >> n >> m;

 	f[0][0]=toBig(1);
 	bigNum kq=toBig(0);
 	FOR(i,1,n)
 		FOR(j,1,n)
 		{
 			f[i][j]=(f[i-1][j-1]+(f[i-1][j]*toBig(j)));
 			kq=ss(kq,f[n][j]);
 		}
 	Print(f[n][m]); 
 	printf("\n"); 
 	Print(kq);
    return 0;
}
