Query on a tree - VNOI

2020-10-07
VNOI Heavy-Light Decomposition , Cây

Link đề bài:

  • https://www.spoj.com/problems/QTREE/.

Hướng dẫn:

Code mẫu:

#include <bits/stdc++.h>
#define ii pair<int, int>
#define fi first
#define se second
using namespace std;
 
const int MAX = 2e4+6;
 
int n, T, cnt, Pos[MAX], Par[MAX], Deep[MAX];
int t[MAX*4], nChild[MAX], Head[MAX];
std::vector<ii> g[MAX];
ii Edges[MAX];
 
void Update(int id, int l, int r, int pos, int val){
	if (l==r){
		t[id] = val;
		return;
	}	
	int mid = (l+r)/2;
	if (pos <= mid) Update(id*2, l , mid, pos, val);
	else Update(id*2+1, mid+1, r, pos, val);
	t[id] = max(t[id*2], t[id*2+1]);
}
 
int ITQuery(int id, int l, int r, int L, int R){
	if (R<l || r<L) return -999999;
	if (L<=l && r<=R){
		return t[id];
	}
	int mid =(l+r)/2;
	return max(ITQuery(id*2, l, mid, L,R), ITQuery(id*2+1, mid+1, r, L, R));
}
 
int Query(int u, int v){
	int res = -999999;
	// di len toi khi nao cung mot nhanh
	while (Head[u]!= Head[v]){
		if (Deep[Head[u]] > Deep[Head[v]]) swap(u, v); // v sau hon u
		res = max(res, ITQuery(1,1,n,Pos[Head[v]], Pos[v]));
		v = Par[Head[v]];
	}
	if (Pos[v] < Pos[u]) swap(u, v);
	res = max(res, ITQuery(1,1,n, Pos[u]+1, Pos[v])); // dinh nghia canh (u,v) nam o vi tri cua dinh v
	return res;
}
 
void DFS(int u, int par){
	nChild[u] = 1;
	int idMx = -1;
	for(int i=0; i< g[u].size(); i++){
		int v = g[u][i].fi;
		if (v == par) continue;
		Par[v] = u;
		Deep[v] = 1 + Deep[u];
		DFS(v, u);
		nChild[u]+= nChild[v];
		if (idMx == -1 || nChild[v] > nChild[g[u][idMx].fi]) idMx = i;
	}
	if (idMx!=-1) swap(g[u][idMx], g[u][0]);
}
 
void HLD(int u, int curHead){
	Head[u] = curHead;
	Pos[u] = ++cnt;
 
	if (g[u][0].fi == Par[u]) return;
	Update(1,1,n,cnt+1, g[u][0].se);
	HLD(g[u][0].fi, curHead);
 
	for(int i=1; i< g[u].size(); i++){
		int v = g[u][i].fi;
		if (v == Par[u]) continue;
		Update(1,1,n,cnt+1, g[u][i].se);
		HLD(v, v);
	}
}
 
void Reset(){
	memset(t, 0, sizeof t);
	cnt = 0;
	for(int i=1; i<=n; i++){
		g[i].clear();
	}
}
 
int main(int argc, char const *argv[])
{
	ios_base::sync_with_stdio(0);
	cin.tie(0); cout.tie(0);
	
	#ifndef ONLINE_JUDGE
	    freopen("input.txt","r",stdin);
	#endif
	
	cin >> T;
	while (T--){
		Reset();
		cin >> n;
		for(int i=1; i<n; i++){
			int u, v, w;
			cin >> u >> v >> w;
			g[u].push_back(ii(v,w));
			g[v].push_back(ii(u,w));
			Edges[i] = ii(u, v);
		}
		DFS(1,1);
		HLD(1,1);
 
		while (1){
			string Type; int u, v;
			cin >> Type;
			if (Type == "DONE") break;
 
			cin >> u >> v;
 
			if (Type == "QUERY"){
				cout << Query(u,v) << "\n";
			}
			else{
				//update
				if (Deep[Edges[u].fi] > Deep[Edges[u].se]) 
					swap(Edges[u].fi, Edges[u].se);
				Update(1,1,n, Pos[Edges[u].se], v);
			}
		}
	}
	return 0;
}
