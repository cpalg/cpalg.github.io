Lockdown

  • Đề bài: lockdown.pdf.
  • Solution: Để phong tỏa 1 con đường mà số con đường bị tăng khoảng cách li' > li là lớn nhất, ta sẽ ưu tiên phong tỏa các con đường đi qua đỉnh 1. Ta thử phong tỏa các con đường đó, sử dụng Dijkstra để đếm xem có bao nhiêu con đường bị tăng khoảng cách.
  • Code mẫu:
#include <bits/stdc++.h>
#define pb push_back
#define FOR(i, a, b) for(int i = int(a); i <= int(b); i++)
#define FORD(i, a, b) for(int i = int(a); i >= int(b); i--)
#define REP(i, r) for(int i = 0; i < r; i++)
#define FAST ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
#define READFILE freopen("lockdown.inp", "r", stdin)
#define WRITEFILE freopen("lockdown.out", "w", stdout);
#define ll long long
#define ii pair<ll, ll>
#define fi first
#define se second
#define ENDL "\n"
using namespace std;

const int MAX=555;

ll n, a[MAX][MAX], d[MAX][MAX], res;

void Dijkstra(int st, int pos){
    for(int i=1; i<=n; i++ ) d[pos][i] = 1e14;
    d[pos][st] = 0;
    priority_queue<ii, vector<ii>, greater<ii> > pq;
    pq.push(ii(0, st));

    while (!pq.empty()){
        int u = pq.top().se, du = pq.top().fi;
        pq.pop();
        if (du != d[pos][u]) continue;
        for(int v=1; v<=n; v++){
            if (d[pos][u] + a[u][v] < d[pos][v]){
                d[pos][v] = d[pos][u] + a[u][v];
                pq.push(ii(d[pos][v], v));
            }
        }
    }
}

int main()
{
        READFILE;
        WRITEFILE;
    #ifndef ONLINE_JUDGE
    #endif
    FAST;

 	cin >> n;
 	FOR(i,1,n)
        FOR(j,1,n){
            cin >> a[i][j];
            if (a[i][j] == -1) a[i][j] = 1e14;
         }
    Dijkstra(1,0);

    FOR(i,1,n){
        ll tmp = a[1][i];
        a[1][i] = 1e14;
        Dijkstra(1,i);
            ll rm = 0;
        FOR(k,1,n){
            if (d[i][k] > d[0][k]) rm++;
        }
        res = max(res, rm);
        a[1][i] = tmp;
    }
    cout << res;
    return 0;
}

QN

  • Đề bài: QN.pdf.
  • Solution: Giả sử đồ thị đã cho có 2^n đỉnh, ta không bỏ đi một đỉnh nào, ta có thể tạo ra được chu trình s1, s2,..., s(2n), s1, vậy để tạo ra đươc chu trình với 2^(n+1) đỉnh, ta chỉ cần đảo ngược thứ tự các đỉnh lại và thêm bit 1 vào đằng trước, thì đồ thị mới này và đồ thị cũ chắc chắn sẽ tạo ra được một chu trình. Mà theo đề bài ta phải xóa một đỉnh k, nên ta chỉ cần xét một đường đi từ đỉnh k+1 về đỉnh k-1 trong chu trình.
  • Code mẫu:

    return 0;
}

EOE

  • Đề bài: EOE.pdf.
  • Solution: Chỉ cần làm theo đề bài nói, nhân trước cộng trừ sau, và xét tính chẵn lẻ.
  • Code mẫu: Code này ngu ko full điểm.
#include <bits/stdc++.h>
#define pb push_back
#define FOR(i, a, b) for(int i = int(a); i <= int(b); i++)
#define FORD(i, a, b) for(int i = int(a); i >= int(b); i--)
#define REP(i, r) for(int i = 0; i < r; i++)
#define FAST ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
#define READFILE freopen("EOE.INP", "r", stdin)
#define WRITEFILE freopen("EOE.OUT", "w", stdout);
#define ll long long
#define ii pair<int, int>
#define fi first
#define se second
#define ENDL "\n"
using namespace std;

const int MAX=-1;

string s, a, b, vt;
int x, y;
int d[10000000];

int main()
{
        READFILE;
        WRITEFILE;
    #ifndef ONLINE_JUDGE
    #endif
    FAST;

 	cin >> s;
 	cin >> a;
 	cin >> b;
 	if (a == "Even") x = 0; else x = 1;
 	if (b == "Even") y = 0; else y = 1;

 	s = "+" + s + "++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++";
 	for(int i=0; i< s.size(); i++){
        if (s[i] == 'x') s[i] = x + '0';
        if (s[i] == 'y') s[i] = y + '0';
        if (s[i] == '-') s[i] = '+';

 	}

 	int i  = 0;
 	d['+'] = 1; d['-']=2; d['*'] = 3;
    for(; i< s.size(); i++){
        if (d[s[i]]){
            vt.push_back(s[i]);
            while (!d[s[i+1]]) i++;
            vt.push_back(s[i]);
        }
    }
    while (d[vt[vt.size()-1]]!= 0) vt.pop_back();
    i = 0;
    int res =0;
    for(; i < vt.size(); i++){
        if (!d[vt[i]]){
            string tmp;
            tmp.push_back(vt[i]);
            if (vt[i+1] == '*') {
                while (i+1 < vt.size() && vt[i+1] != '+') {
                    tmp.push_back(vt[i+1]);
                    i++;
                }
            }
            if (tmp.size() == 1){
                res+= tmp[0] - '0';
            }
            else{
                ll kq = 1;
                FOR(k, 0, tmp.size()-1){
                    if (!d[tmp[k]]) kq*= (tmp[k] - '0'), kq%=2;
                }
                res+= kq%2;
            }
        }
    }
        if (res&1) cout <<"Odd";
        else cout << "Even";
    return 0;
}

PGIFT

  • Đề bài: PGIFT.pdf.

  • Solution: Vì chỉ có bốn hình chữ nhật, nên ta có thể duyệt và tham lam xét tất cả các trường hợp đặt hình chữ nhật:

    • Ta xét tất cả các hoán vị thứ tự đặt của các hình chữ nhật, và ta xét trường hợp đặt dọc hay quay ngang một hình chữ nhật nào đó.
    • Ta biểu diễn một hình chữ nhật bằng 2 cặp tọa độ trái trên và phải dưới như đề bài (x_i, y_i) và (u_i, v_i).
    • Ta nhận xét, một hình chữ nhật thứ i mới được đặt vào mà chi phí tối ưu thì tọa độ trái trên của nó phải là (x_i, y_i) với x_i thuộc các {0, u_1, u_2,..., u_(i-1)} và y_i thuộc {0,v_1, v_2,..., v_(i-1)}. Vì chỉ có bốn hình nên sẽ có không quá 16 lần xét.
    • Sau khi ta đặt hình chữ nhật thứ i vào ta thêm (u_i, v_i) vào 2 tập hợp trên.

  • Code mẫu:

#include <iostream>
#include <algorithm>
#include <vector>
#define int int64_t
#define ii pair<int, int>
#define iiii pair < pair<int, int>,  pair<int, int> >
#define fi first
#define se second
using namespace std;

ii a[87];
int t[558], res = 2e18+6;
bool vs[555];

bool cat(iiii a, iiii b){
	if (max(a.fi.fi, b.fi.fi) >= min(a.se.fi, b.se.fi)) return false;
	if (max(a.fi.se, b.fi.se) >= min(a.se.se, b.se.se)) return false;
	return true;
}

void Cal(int id, vector<int> X, vector<int> Y, vector<iiii> rect){
    for(int i=0; i< X.size(); i++)
        for(int j=0; j< Y.size(); j++){
            bool ok = true;
            for(int r=0; r< rect.size(); r++){
                if (cat(rect[r], iiii(ii(X[i], Y[j]), ii(X[i]+a[t[id]].fi,Y[j]+a[t[id]].se) ))){
                    ok = false;
                }
            }
            if (!ok) continue;
            rect.push_back(iiii(ii(X[i], Y[j]), ii(X[i]+a[t[id]].fi,Y[j]+a[t[id]].se)));
            X.push_back(X[i]+a[t[id]].fi);
            Y.push_back(Y[j]+a[t[id]].se);
            if (id < 4) Cal(id+1, X, Y, rect);
            else{
                res =min(1ll* (*max_element(X.begin(), X.end()))* (*max_element(Y.begin(), Y.end())),res);
                // cout << res << endl;
                return;
            }
            X.pop_back();
            Y.pop_back();
            rect.pop_back();

        }
}

void Quay(int id){
	if (id > 4) {
		std::vector<int> X;
		X.push_back(0ll);
		std::vector<int> Y;
		Y.push_back(0ll);
		std::vector<iiii> rect;
		Cal(1, X, Y, rect);	
		return;
	}
	for(int j=1; j<=4; j++){
		if (!vs[j]){
			vs[j] = true;
			t[id] = j;
			Quay(id+1);
			vs[j] = false;
		}
	}
}

bool Bit(int mask, int pos){
	return mask & (1 << (pos-1));
}

signed main(int argc, char const *argv[])
{
	freopen("PGIFT.INP","r",stdin);
	freopen("PGIFT.OUT","w",stdout);
	for(int i=1; i<=4; i++) cin >> a[i].fi >> a[i].se;
	for(int mask=0; mask < (1<<4); mask++){
		for(int i=1; i<=4; i++) {
			if (Bit(mask,i)) swap(a[i].fi , a[i].se);
			t[i] = 0;
		}
		Quay(1);
		for(int i=1; i<=4; i++) {
			if (Bit(mask,i)) swap(a[i].fi , a[i].se);
			t[i] = 0;
		}
	}
	cout << res;
	return 0;
}

SQSORT

  • Đề bài: SQSORT.pdf.

  • Solution: Đề bài cho k<=2^n nên ta có thể áp dụng chia để trị, theo đề bài mỗi stack và queue xen kẽ nhau, nên ta có thể sử dụng một cặp stack và queue để tách dữ liệu làm 2 tập, tương tự như quicksort, và đưa lần lượt 2 tập này về đích.

  • Code mẫu:

#include <algorithm>
#include <iostream>
#include <vector>
#define ii pair<int, int>
#define fi first
#define se second
using namespace std;

const int MAX = 1e6+1e5+6;

int n, k, b[MAX];
ii a[MAX];
std::vector<int> ans;

void Cal(int id, int l, int r, std::vector<int> rs){
	int mid = (l+r)/2;
	if (rs.size()<=1){
		for(int i=id; i<= 2*n; i++) ans.push_back(i);
		return;
	}
	std::vector<int> rs1;
	std::vector<int> rs2;
	for(int i=0; i< (int)rs.size(); i++){
		if (rs[i] <= mid){
			rs1.push_back(rs[i]);
			ans.push_back(id); ans.push_back(id+1);
		}
		else {
			rs2.push_back(rs[i]);
			ans.push_back(id);
		}
	}
	Cal(id+2, l, mid, rs1);
	reverse(rs2.begin(), rs2.end());
	for(int i=0; i< (int)rs2.size(); i++) ans.push_back(id+1);
	Cal(id+2, mid+1,r , rs2);
}

int main(int argc, char const *argv[])
{
	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
	freopen("SQSORT.INP","r", stdin);
	freopen("SQSORT.OUT","w", stdout);
	cin >> n >> k;
	for(int i=1; i<=k; i++){
		cin >> a[i].fi;
		a[i].se = i;
	}
	sort(a+1, a+k+1);
	for(int i=1; i<=k; i++){
		b[a[i].se] = i;
	}
	std::vector<int> rr;
	for(int i=1; i<=k; i++) rr.push_back(b[i]);

	Cal(1, 1, k, rr);
	
	cout << ans.size() << endl;
	for(int i=0; i< (int)ans.size(); i++) cout << ans[i] <<  " ";
	return 0;
}

BST

  • Đề bài: BST.pdf.

  • Solution:

    • Không tạo được cây tìm kiếm nhị phân độ cao h khi và chỉ khi số nút >= 2^h.
    • Ta có nhận xét, nút gốc của cây là số đầu tiên trong dãy hoán vị, vậy ta có thể tối ưu nút gốc của cây tham lam bằng cách đặt các nút lớn hơn nút gốc vào bên phải, vậy ta có thể xác định nút nào sẽ là nút gốc nhỏ nhất, nút gốc nhỏ nhất chính là kích thước của cây nhị phân đầy đủ bên phải - 1.
    • Từ đó có thể đệ quy cho các cây con.

  • Code mẫu:

#include <bits/stdc++.h>
#define pb push_back
#define FOR(i, a, b) for(int i = int(a); i <= int(b); i++)
#define FORD(i, a, b) for(int i = int(a); i >= int(b); i--)

#define FAST ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

#define WRITEFILE freopen("BST.OUT", "w", stdout);

#define se second
#define ENDL "\n"

const int MAX=-1;

int n, h;

void Build( int l, int r, int h){
    int mid =  max(l, r- (1<<(h-1))+1);
    cout << mid << " ";	
    if (l==r) return;
    if (l< mid) Build(l, mid-1, h-1);
    if (mid < r) Build(mid+1, r, h-1);
}

int main()
{
    #ifndef ONLINE_JUDGE
        READFILE;
        WRITEFILE;
    #endif
    FAST;

 	cin >> n >> h;
 	if (n >= (1<<h)){
        cout << -1;
        return 0;
 	}
    Build(1,n,h);
    return 0;
}
