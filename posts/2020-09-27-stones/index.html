#include <bits/stdc++.h>
#define pb push_back
#define FOR(i, a, b) for(int i = int(a); i <= int(b); i++)
#define FORD(i, a, b) for(int i = int(a); i >= int(b); i--)
#define REP(i, r) for(int i = 0; i < r; i++)
#define FAST ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
#define READFILE freopen("input.txt", "r", stdin)
#define WRITEFILE freopen("output.txt", "w", stdout);
#define ll long long
#define ii pair<int, int>
#define fi first
#define se second
#define ENDL "\n"
using namespace std;

const int MAX= 111;
const int MAXK = 1e5+6;

int a[MAX], n, k;
int f[MAXK][2];

int DP(int len, bool player){
    if (len < 0) return 0;
    if (f[len][player]!= -1) return f[len][player];
    {
        bool tmp = 1;
        for(int i=1; i<=n; i++){
            tmp &= DP(len - a[i], 1- player);
            if (!tmp) break;
        }
        f[len][player] = 1- tmp;
        return (1- tmp);
    }
}

int main()
{
    #ifndef ONLINE_JUDGE
        READFILE;
    #endif
    FAST;

 	cin >> n >> k;
 	for(int i=1; i<=n; i++){
        cin >> a[i];
 	}
 	sort(a+1, a+n+1);
 	memset(f,-1,sizeof f);
 	int res = DP(k,0);
 	if (!res) cout << "First";
 	else cout << "Second";
    return 0;
}
