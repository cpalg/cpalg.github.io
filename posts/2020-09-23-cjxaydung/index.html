Xây dựng vùng LS Vagos - LQDOJ.

2020-09-23
LQDOJ Đồ thị , Khớp, cầu , DFS , Lý thuyết

Link đề bài:

  • http://lqdoj.edu.vn/problem/cjxaydung.

Hướng dẫn:

  • Đây là bài toán tìm cầu cơ bản, với mỗi cạnh u,v ta đang xét với u là cha của v, nếu từ v chỉ đi được các đỉnh thấp hơn u thì (u,v) là cầu.

Code mẫu:

#include <bits/stdc++.h>
#define pb push_back
#define FOR(i, a, b) for(int i = int(a); i <= int(b); i++)
#define FORD(i, a, b) for(int i = int(a); i >= int(b); i--)
#define REP(i, r) for(int i = 0; i < r; i++)
#define FAST ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
#define READFILE freopen("input.txt", "r", stdin)
#define WRITEFILE freopen("output.txt", "w", stdout);
#define ll long long
#define ii pair<int, int>
#define fi first
#define se second
#define ENDL "\n"
using namespace std;

const int MAX= 3e5+6;

int Low[MAX], Num[MAX], cnt, m, n, bridge[MAX],res;
vector<ii> g[MAX];

void DFS(int u, int  par){
    Low[u] = Num[u] = ++cnt;
    for(ii t: g[u]){
        int v = t.fi, id = t.se;
        if (v==par) continue;
        if (Num[v]) Low[u] = min(Low[u], Num[v]);
        else{
            DFS(v,u);
            Low[u] = min(Low[u], Low[v]);
            if (Low[v] > Num[u]){
                bridge[id] = true;
                res++;
            }
        }
    }
}

int main()
{
    #ifndef ONLINE_JUDGE
        READFILE;
    #endif
    FAST;

 	cin >> n >> m;
 	FOR(i,1,m){
        int u, v;
        cin >> u >> v;
        g[u].push_back(ii(v,i));
        g[v].push_back(ii(u,i));
 	}
 	FOR(i,1,n){
        if (!Num[i]) DFS(i,i);
 	}
 	cout<< res << ENDL;
 	FOR(i,1,n) if (bridge[i]) cout << i << " ";
    return 0;
}
