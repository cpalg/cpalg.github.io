CJ và Catalina - LQDOJ.

2020-09-24
LQDOJ Đồ thị , Đường đi ngắn nhất , DFS , Quy hoạch động trạng thái

Link đề bài:

  • http://lqdoj.edu.vn/problem/cjandcatalina.

Hướng dẫn:

  • Đầu tiên ta dùng thuật toán Dijkstra để tìm đường đi ngắn nhất giữa mọi cặp đỉnh trong k cặp đề bài cho. (\( k <=20 \)).
  • Sử dụng quy hoạch động trạng thái:
    • Gọi f[mask][i] là đường đi ngắn nhất đi qua tất cả các đỉnh thuộc trạng thái mask, và kết thúc tại đỉnh i
    • Trạng thái ban đầu: f[1 << i][i] = 0
    • Ta xét mỗi trạng thái và cập nhật trạng thái như sau: f[j][newmask] = min(f[j][newmask], f[i][mask] + d[i][j])

Code mẫu:

#include <bits/stdc++.h>
#define FAST ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
#define ll long long
#define ii pair<long long, long long>
#define fi first
#define se second
#define ENDL "\n"
using namespace std;

const int MAX= 200005;

long long d[20][MAX], a[20], n, m, k, f[524289][22];
vector<ii> g[MAX];
unordered_map<long long,long long> mp;

void Dijkstra(int t){
    int st = a[t]; int cnt =0;
    for(int i=1; i<=n; i++) d[t][i] = 1000000000000;
    d[t][st] = 0;
    priority_queue <ii, vector<ii>, greater<ii> > pq;
    pq.push(ii(0,st));
    while (!pq.empty()){
        ll u = pq.top().se, du = pq.top().fi; pq.pop();
        if (d[t][u] != du) continue;
        if (mp[u] > 0) cnt++;
        if (cnt ==k) return;

        for(ii tmp: g[u]){
            long long uv = tmp.fi, v= tmp.se;
            if (d[t][u] + uv < d[t][v]){
                d[t][v] = d[t][u] + uv;
                pq.push(ii(d[t][v],v));
            }
        }
    }
}

bool getBit(long long mask, long long pos){
    return mask & (1<<(pos-1));
}

void Solve(){
    long long MSK = (1 << k) - 1;
    for(int i=1; i<= MSK; i++) for(int j=1; j<=k; j++) f[i][j] = 1000000000000;
    for(int i=1; i<=k; i++) f[1 << (i-1)][i] = 0;

    for(int msk=1; msk<= MSK; msk++)
        for(int i=1; i<=k; i++)
            if (!getBit(msk, i)){
                for(int j=1; j<=k; j++){
                    if (getBit(msk,j)){
                        f[msk|(1<<(i-1))][i] = min(f[msk|(1<<(i-1))][i], f[msk][j] + d[i][a[j]]);
                    }
                }
            }
    long long res = 1000000000000;
    for(int i=1; i<=k; i++) res = min(res,f[MSK][i]);
    cout << res;
}

signed main()
{
    FAST;

 	cin >> n >>  m >> k;
 	for(int i=1; i<=k; i++) cin >> a[i];
 	for(int i=1; i<=m; i++){
        long long u, v, dd;
        cin >> u >> v >> dd;
        g[u].push_back(ii(dd,v));
        g[v].push_back(ii(dd,u));
 	}
 	for(int i=1; i<=k; i++) mp[a[i]] = i;
    for(int i=1; i<=k; i++) {
        Dijkstra(i);
    }
    Solve();
    return 0;
}
