Tìm bao lồi trong mặt phẳng - Lý thuyết

2020-09-23
Toán Hình học , Lý thuyết

Định nghĩa:

Trong hình học tính toán (computational geometry), bao lồi (convex hull) của một tập điểm là tập lồi nhỏ nhất (theo diện tích, thể tích, …) mà tất cả các điểm đều nằm trong tập đó.

Thuật toán Graham:

  • Chọn đỉnh dưới trái cùng làm gốc, đỉnh này chắc chắn thuộc bao lồi.

  • Sắp xếp các đỉnh còn lại theo chiều kim đồng hồ tính từ đỉnh đó ( các đỉnh nằm trong một nửa mặt phẳng) (Nếu 2 đỉnh này cùng một tạo một góc so với đỉnh trái cùng thì ta xếp theo xa dần).

  • Ta xét lần lượt các đỉnh theo thứ tự, 2 đỉnh cuối cùng của tập là p3 và p2, đỉnh đang xét là p1, xét 2 vector v1(p3,p2) và v2(p2,p1), Nếu vector v2 hướng theo chiều kim đồng hồ so với v1 thì có thể chấp nhận điểm đó, còn nếu 2 vectỏr thẳng hàng hoặc hướng ngược chiều kim đồng hồ thì điểm p2 chắc chắn không thuộc bao lồi, nên ta lấy điểm p2 ra khỏi tập.

  • Có thể dùng tích chéo để xét chiều 2 vector(tham khảo tại đây: Link).

    1. Nếu \(\vec{u} * \vec{v} < 0 \) thì \(\vec{v} \) nằm theo chiều kim đồng hồ so với \(\vec{u} \) .
    2. Nếu \(\vec{u} * \vec{v} = 0 \) thì \(\vec{u} \), \(\vec{v} \) thẳng hàng.
    3. Ngược lại thì \(\vec{v} \) nằm ngược chiều kim đồng hồ so với \(\vec{u} \).

#include <bits/stdc++.h>
#define pb push_back
#define FOR(i, a, b) for(int i = int(a); i <= int(b); i++)
#define FORD(i, a, b) for(int i = int(a); i >= int(b); i--)
#define REP(i, r) for(int i = 0; i < r; i++)
#define FAST ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
#define READFILE freopen("input.txt", "r", stdin)
#define WRITEFILE freopen("output.txt", "w", stdout);
#define ll long long
#define ii pair<ll, ll>
#define fi first
#define se second
#define x first
#define y second
using namespace std;

ii root; // = d[0] de de sap xep

int getMin(int i, int j, vector<ii> &d){
	if (d[i].x == d[j].x){
		if (d[i].y < d[j].y) return i;
		else return j;
	}
	if (d[i].x < d[j].x) return i;
	return j;
}

ii nextToTop(stack<ii> &st){
	ii tmp = st.top();
	st.pop();
	ii ret = st.top();
	st.push(tmp);
	return ret;
}

ll Mul(ii a, ii b, ii c){ // Tich cheo
	ii v1 = ii(b.fi - a.fi, b.se - a.se);
	ii v2 = ii(c.fi - b.fi, c.se - b.se);
	return v1.x * v2.y - v2.x * v1.y;
}
 
double kCach(ii a, ii b)
{
	return hypot(a.x-b.x, a.y - b.y);
}

bool cmp(ii a, ii b){
	ii v1 = ii(a.x - root.x, a.y - root.y);
	ii v2 = ii(b.x - root.x, b.y - root.y);
	if (v1.x * v2.y == v1.y * v2.x){// neu 2 diem cung kcach thi sap theo xa dan
		return kCach(root,a) < kCach(root,b);
	}
	return (v1.x * v2.y - v1.y * v2.x) < 0;
}

int main()
{
    #ifndef ONLINE_JUDGE
        READFILE;
    #endif
    FAST;
    
 	int n;
 	while (cin >> n){
 		if (n==0) break;

 		int pMin = 0;
 		set<ii> S;

 		for(int i =0 ;i<n; i++){
 			ii tmp;
 			cin >> tmp.x >> tmp.y;
 			S.insert(tmp);
 		}

 		vector<ii> d(S.begin(), S.end());
 		n = d.size();

 		for(int i=0; i<n; i++){
 			pMin = getMin(pMin, i, d); // chon dinh trai duoi cung chac chan se thuoc bao loi
 		}
 		swap(d[pMin], d[0]);
 		root = d[0]; 		
 		sort(d.begin()+1,d.end(), cmp); // sap xep cac diem theo chieu kim dong ho 

 		stack<ii> st;
 		if (n>0) st.push(d[0]);
 		if (n>1) st.push(d[1]);

 		for(int i=2; i<n; i++){ // xet lan luot cac diem theo chieu kim dong ho
	 		while (st.size() >= 2 && Mul(nextToTop(st), st.top(), d[i]) >= 0ll) st.pop();
	 		// Neu 3 diem cuoi cung tao duong lom (re trai)
	 		st.push(d[i]);
	 	}

	 	cout << st.size() << "\n";
	 	while (!st.empty()){
	 		cout << st.top().x << " " << st.top().y << "\n";
	 		st.pop();
	 	}
 	}
    return 0;
}
