SGAME8 - LQDOJ

#include <bits/stdc++.h>
#define ii pair<int, int>
#define fi first
#define se second
using namespace std;
typedef vector<int> bigNum;

const int MAX = -1;
const int BASE = 1000000000;

bigNum Fix(bigNum a){
	a.push_back(0);
	for(int i=0; i<(int)a.size(); i++){
		a[i+1]+= a[i]/BASE;
		a[i]%= BASE;
		while (a[i] < 0) a[i]+= BASE, a[i+1]--;
	}
	while (a.size()>1 && a.back()==0) a.pop_back();
	return a;
}

void Print(bigNum a){
	printf("%d", a.back());
	for(int i=a.size()-2; i>=0; i--){
		printf("%09d", a[i]);
	}
}

bigNum operator + (bigNum a, bigNum &b){
	a.resize(max(a.size(), b.size()));
	for(int i=0; i<(int)b.size(); i++) a[i]+= b[i];
	return Fix(a);
}

int main(int argc, char const *argv[])
{
	ios_base::sync_with_stdio(0);
	cin.tie(0); cout.tie(0);
	
	#ifndef ONLINE_JUDGE
	    freopen("input.txt","r",stdin);
	#endif
	string s;
	cin >> s;
	s = ' ' + s;
	bigNum st, num;
	st.push_back(1); num.push_back(1);
	for(int i=1; i< (int)s.size(); i++){
		if (s[i] == '*'){ // co them 3 cach
			st= st+st+st+st+st+num; 
			num= num+num+num;
		}
		else if (s[i] == 'L'){
			st= st+st;
		}
		else if (s[i]== 'R') st = st+st+num;
	}
	Print(st);
	return 0;
}
