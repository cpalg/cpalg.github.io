Land Acquisition - VNOI

2020-10-03
Quy hoạch động bao lồi Quy hoạch động , VNOI

Link đề bài:

  • https://www.spoj.com/problems/ACQUIRE/.

Hướng dẫn:

  • Ta có một nhận xét rằng: Giả sử tồn tại hai hình chữ nhật với kích thước \((a,b)\) và \((x,y)\). nếu \(a<=x\) và \(b<=y \) thì ta có thể bỏ luôn hình chữ nhật kích thước \((a,b)\) bởi vì chỉ cần gộp hình chữ nhật kích thước \((a,b)\) vào hình chữ nhật kích thước \((x,y)\) thì sẽ không mất phí, do đó ta loại bỏ tất cả các hình chữ nhật thừa.
  • Ta tiến hành sắp xếp các hình chữ nhật tăng dần theo chiều dài \(-> \) chiều rộng các hình chữ nhật sẽ giảm dần.
  • Ta có nhận xét sau: Để gộp các hình chữ nhật với chi phí tối ưu thì ta phải gộp các hình chữ nhật liên tiếp theo thứ tự sắp xếp lại với nhau, bởi vì các gộp này chỉ mất phí bằng tích của chiều rộng hình bên trái cùng và chiều dài hình bên phải cùng, do đó bài toán quy về bài toán phân hoạch thành các đoạn sao cho tổng của các tích chiêu dài và chiều rộng là nhỏ nhất.
  • Ta có thể sử dụng phương pháp quy hoạch động để giải bài toán này: \(f_i = max( f_j + h_{j+1} * w_i)\) với \((0 <= j < i)\) Ta thấy đây là đồ thị của phương trình bậc 1 \(y_j = a*x + b\).
  • Ta có thể duyệt trâu với ĐPT \(\mathcal{O}(N^2)\), nhưng để không vượt quá giới hạn thời gian cho phép ta sử dụng phương pháp quy hoạch động bao lồi tìm đường thẳng \(j\) mà \(f_j + h_{j+1} * w_i\) là lớn nhất.

Code mẫu:

#include <bits/stdc++.h>
#define pb push_back
#define FOR(i, a, b) for(int i = (a); i <= (b); i++)
#define FORD(i, a, b) for(int i = (a); i >= int(b); i--)
#define REP(i, r) for(int i = 0; i < r; i++)
#define FAST ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
#define READFILE freopen("input.txt", "r", stdin)
#define WRITEFILE freopen("output.txt", "w", stdout);
#define ll long long
#define ii pair<ll, ll>
#define x first
#define y second
using namespace std;

const int MAX= 1e5+6;

std::vector<int> A;
std::vector<int> B;
std::vector<ii> Rect;
int n, m, Pointer;
ii a[MAX];
 
bool Bad(int l1, int l2, int l3){
	return (1ll*B[l3]-B[l1])*(A[l1]-A[l2])<1ll*(B[l2]-B[l1])*(A[l1]-A[l3]);
}

void Add(ll a, ll b){ // them duong thang y = a*x + b
	A.push_back(a); B.push_back(b);
	while (A.size() >=3 && Bad(A.size()-3, A.size()-2, A.size()-1)){
		A.erase(A.end()-2);
		B.erase(B.end()-2);
	}
}

ll Query(ll x){
	if (Pointer >=A.size()) Pointer = A.size()-1;
	while (Pointer < A.size() -1 &&   
		A[Pointer+1]*x+B[Pointer+1]<A[Pointer]*x+B[Pointer]) Pointer++;
	return 1ll*A[Pointer]*x+B[Pointer];
}

int main()
{
    #ifndef ONLINE_JUDGE
        READFILE;
    #endif
    FAST;
    
 	cin >> m;
 	FOR(i,1,m){
 		cin >> a[i].x >> a[i].y;
 	}
 	sort(a+1, a+m+1);

 	FOR(i,1,m){
 		while (Rect.size() && Rect.back().y <= a[i].y) Rect.pop_back();
 		Rect.push_back(a[i]); 
 	}
 	n = Rect.size();

 	Add(Rect[0].y, 0);

 	ll Cost;
 	REP(i,n){
 		Cost = Query(Rect[i].x);
 		if (i < n-1) Add(Rect[i+1].y, Cost);
 	}
 	cout << Cost;
    return 0;
}
