Swap Game - CSES

2020-10-17
CSES BFS

Link đề bài:

  • https://cses.fi/problemset/task/1670/.

Hướng dẫn:

  • Tạo đồ thị mỗi đỉnh chính là mỗi trạng thái của bảng, có 9! trạng thái, mỗi cạnh chính là một bước di chuyển 2 ô kề nhau. Sau đó BFS từ đỉnh gốc đề bài cho đến đỉnh đích (123456789).

Code mẫu:

#include <unordered_map>
#include <queue>
#include <vector>
#include <iostream>
#include <algorithm>
#include <string.h>
#define ii pair<int, int>
#define fi first
#define se second
using namespace std;

string s;
std::vector<ii> g;

int main(int argc, char const *argv[])
{
	ios_base::sync_with_stdio(0);
	cin.tie(0); cout.tie(0);
	
	#ifndef ONLINE_JUDGE
	    // freopen("input.txt","r",stdin);
	#endif
	
	
	for(int i=1 ; i<=9; i++){
		char tmp;
		cin >> tmp;
		s.push_back(tmp);
	}
	unordered_map<string , int> d;
	string t = "123456789";
	for(int i=0; i<3; i++){
		for(int j=0; j<2; j++){
			g.push_back(ii(j+i*3, j+1+i*3));
			g.push_back(ii(i+j*3, i+(j+1)*3));
		}
	}
	queue<string> q;
	q.push(s);
	d[s] = 1;
	while(!q.empty()){
		string v = q.front(), u = q.front(); q.pop();
		if (u==t) break;
		for(ii x: g){
			swap(v[x.fi], v[x.se]);
			if (!d[v]) {
				q.push(v);
				d[v]=d[u]+1;
			}
			swap(v[x.fi], v[x.se]);
		}
	}
	cout << d[t]-1;
	return 0;
}```
