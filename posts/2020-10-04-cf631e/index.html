Product Sum - Codeforces

2020-10-04
Quy hoạch động bao lồi Quy hoạch động , Codeforces

Link đề bài:

  • https://codeforces.com/problemset/problem/631/E.

Hướng dẫn:

  • Ta xét 2 trường hợp:

    1. TH1: Di chuyển một khối từ vị trí \(i\) tới vị trí \(j\) \((i < j)\).

      • Tất cả các khối ở vị trí thứ \(k\) \((i+1 <= k <= j)\) sẽ bị giảm đi \(a_k\).
      • Giá trị của khối \(i\) sẽ tăng từ \(i * a_i\) lên thành \(j * a_j\).
      • Ta có công thức qhđ: \(dp_j = c - \sum\limits_{k = i+1}^j {a_k} + j * a_i - i * a_i\).
      • Ta thấy hàm này có dạng \(a*x + b\) với tham số \(j\).

    2. TH2: Di chuyển một khối từ vị trí \(i\) tới vị trí \(j\) \((i > j)\).

      • Ta làm tương tự trường hợp trên.

Code mẫu:

#include <bits/stdc++.h>
#define pb push_back
#define FOR(i, a, b) for(int i = int(a); i <= int(b); i++)
#define FORD(i, a, b) for(int i = int(a); i >= int(b); i--)
#define REP(i, r) for(int i = 0; i < r; i++)
#define FAST ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
#define READFILE freopen("input.txt", "r", stdin)
#define WRITEFILE freopen("output.txt", "w", stdout);
#define ll long long
#define ii pair<long double, long double>
#define fi first
#define se second
using namespace std;

const int MAX= 4e5+6;

int n;
ll a[MAX], f[MAX], s[MAX], sum;
std::vector<ii> H;
 
bool Bad(int l1, int l2, int l3){
	return (H[l3].se-H[l1].se)*(H[l1].fi-H[l2].fi)<(H[l2].se-H[l1].se)*(H[l1].fi-H[l3].fi);
}

void Add(ll a, ll b, bool ng = 0){
	H.push_back(ii(a,b));
	while (H.size() >= 3 && ng != Bad(H.size()-1, H.size()-2, H.size()-3)) {
		H.erase(H.end()-2);
	}
}

ll Query(ll x){
	int l = 0, r = H.size()-1;
	if (l>r) return -1e17;

	while (l<r){
		int mid =  l+ (r-l)/2;
		if (H[mid].fi * x + H[mid].se <= H[mid+1].fi * x + H[mid+1].se){
			l = mid + 1;
		}
		else r = mid;
	}
	return H[l].fi *x + H[l].se;
}

int main()
{
    #ifndef ONLINE_JUDGE
        READFILE;
    #endif
    FAST;
    
 	cin >> n;
 	FOR(i,1,n){
 		cin >> a[i];
 		sum+=i*a[i];
 		s[i] = s[i-1]+a[i];
 	}
 	ll res = sum;
 	FORD(i,n,1){
 		f[i] = sum + s[i] - i*a[i] + Query(a[i]);
 		res = max(res, f[i]);
 		Add(i, -s[i]);
 	}
 	H.clear();
 	FOR(i,1,n){
 		f[i] = sum + s[i-1] - i*a[i] + Query(a[i]);
 		res = max(res, f[i]);
 		Add(i, -s[i-1],1);
 		if (f[i] == -2) cout << i << " ";
 	}
 	cout << res;
    return 0;
}
