#include <iostream>
#include <algorithm>
#include <vector>
#include <string.h>
#define int unsigned long long
#define ii pair<int, int>
#define fi first
#define se second
using namespace std;

const int MAX = 355;

int n, Prime[MAX], f[MAX][MAX], r[MAX], T;
std::vector<int> pr;

void Eratos(){
	memset(Prime, 1, sizeof Prime);
	Prime[0] = Prime[1] = 0;
	for(int i=2; i*i < MAX; i++){
		if (Prime[i]) for(int j=i*2; j<MAX; j+=i){
			Prime[j] = false;
		}
	}
	pr.push_back(0);
	for(int i=1; i<MAX; i++) if (Prime[i]) pr.push_back(i);
}

signed main(int argc, char const *argv[])
{
	ios_base::sync_with_stdio(0);
	cin.tie(0); cout.tie(0);
	
	#ifndef ONLINE_JUDGE
	    freopen("input.txt","r",stdin);
	#endif
	
	cin >> T;
	Eratos();
	f[0][0] = 1;
	for(int i=1; i<pr.size(); i++){
		for(int j=0; j<MAX; j++){
			f[i][j] = f[i-1][j];
			if (j>0) f[i][j] = max(f[i][j], f[i][j-1]);
			for(int k=pr[i]; k<=j; k*=pr[i]){
				f[i][j] = max(f[i][j], f[i-1][j-k]*k);
			}
		}
	}


	while (T--){
		cin >> n;
		cout << f[pr.size()-1][n] << "\n";
	}
	return 0;
}```
