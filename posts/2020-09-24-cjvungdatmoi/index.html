CJ và vùng đất mới - LQDOJ.

2020-09-23
LQDOJ Đồ thị , Khớp, cầu , Thành phần song liên thông , Lý thuyết , DFS

Link đề bài:

  • http://lqdoj.edu.vn/problem/cjvungdatmoi.

Hướng dẫn:

  • Đây là bài toán tìm thành phần song liên thông (biconnected component).
  • Thành phần song liên thông là một thành phần liên thông mà khi ta bỏ một đỉnh bất kì nào, thì các đỉnh còn lại vẫn cùng thuộc một thành phần liên thông.
  • Ta có một nhận xét là các thành phần song liên thông được nối với nhau bằng các khớp.
  • Để giải quyết bài này ta sử dụng thuật toán tarjan để tìm các đỉnh khớp, nếu một đỉnh là khớp thì các cạnh ở phía bên dưới khớp đều thuộc cùng một thành phần liên thông mạnh. Nên ta sẽ lấy các cạnh bên dưới đỉnh khớp đó ra. Nếu đồ thị không có khớp thì tất cả các đỉnh cùng thuộc một thành phần song liên thông. (Lưu ý rằng ta chỉ xét các cạnh nằm trong thành phần song liên thông bởi vì mỗi cạnh chỉ thuộc duy nhất một thành phần song liên thông, tuy nhiên mỗi đỉnh có thể thuộc nhiều thành phần song liên thông(Hình trên)).

#include <bits/stdc++.h>
#define pb push_back
#define FOR(i, a, b) for(int i = int(a); i <= int(b); i++)
#define FORD(i, a, b) for(int i = int(a); i >= int(b); i--)
#define REP(i, r) for(int i = 0; i < r; i++)
#define FAST ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
#define READFILE freopen("input.txt", "r", stdin)
#define WRITEFILE freopen("output.txt", "w", stdout);
#define ll long long
#define ii pair<int, int>
#define fi first
#define se second
#define ENDL "\n"
using namespace std;

const int MAX= 1e5+6;

int n, m, Num[MAX], Low[MAX], cnt, mxSz, Id[MAX],nComp;
stack<ii> st;
vector<int> g[MAX];
vector<int> res;

void DFS(int u, int par){
    Num[u] = Low[u] = ++cnt;
    for(int v: g[u]){
        if (v==par) continue;
        if (Num[v]) Low[u] = min(Low[u], Num[v]);
        else{
            st.push(ii(u,v));
            DFS(v,u);
            Low[u] = min(Low[u], Low[v]);
            if (Low[v] >= Num[u]){
                int sz = 1;
                nComp++;
                ii k;
                vector<int> tmp;
                do{
                    k = st.top(); st.pop();
                    tmp.push_back(k.se);

                } while (k != ii(u,v));
                tmp.push_back(u);
                if (tmp.size() > mxSz) mxSz = tmp.size(), res = tmp;
            }
        }
    }
}

int main()
{
    #ifndef ONLINE_JUDGE
        READFILE;
    #endif
    FAST;

 	cin >> n >> m;
 	for(int i=1; i<=m; i++){
        int u , v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u);
 	}
 	for(int i=1; i<=n; i++) if (!Num[i]) DFS(i,i);
 	cout << res.size() << ENDL;
 	for(int i=0; i< res.size(); i++) cout << res[i] << " ";
    return 0;
}
