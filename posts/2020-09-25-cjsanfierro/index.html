CJ tới San Fierro - LQDOJ.

2020-09-24
LQDOJ Đồ thị , Đường đi ngắn nhất , DFS , Cây khung nhỏ nhất , Tìm kiếm nhị phân

Link đề bài:

  • http://lqdoj.edu.vn/problem/cjsanfierro.

Hướng dẫn:

  • Đầu tiên ta tìm tuyến đường có độ cao cho phép lớn nhất, có thể tìm bằng nhiều cách: Dijkstra, DFS kết hợp với tìm nhị phân, xây dựng cây khung nhỏ nhất..
  • Giả sử độ cao tối đa là H, để tìm kiếm con đường có độ cao H và ngắn nhất, thì ta chỉ xét những con đường có độ cao >= H (Những con đường còn lại sẽ làm cho độ cao nhỏ hơn H), Dijkstra tìm đường đi ngắn nhất trên tập cạnh mới này, ta thu được kết quả bài toán.

Code mẫu:

#include <bits/stdc++.h>
#define ll long long
#define ii pair<ll, ll>
#define fi first
#define se second
using namespace std;

const int MAX = 2e5+6;

struct Data{
    ll v, d, h;
    Data(ll vv= 0, ll dd = 0, ll hh =0){
         v = vv; d= dd; h = hh;
    }
};

ll n, m, vs[MAX], dem, LM, st, en, ok, d[MAX], Pre[MAX];
vector<Data> g[MAX];

void DFS(ll u, ll Lim){
    vs[u] = dem;
    if (u == en){
        ok = true;
        return;
    }
    for(Data t: g[u]){
        if (t.h < Lim) continue;
        ll v = t.v;
        if (vs[v]!= dem) {
            DFS(v, Lim);
        }
    }
}

void Trace(ll u){
    stack<ll> st;
    while (u>0){
        st.push(u);
        u = Pre[u];
    }
    while (st.size()) cout << st.top() << " ", st.pop();
}

void Dijkstra(){
    for(int i=1; i<=n; i++) d[i] = 1e16;
    d[st] =0;
    priority_queue<ii, vector<ii>, greater<ii> > pq;
    pq.push(ii(0,st));
    while (!pq.empty()){
        ll u = pq.top().se, du = pq.top().fi; pq.pop();
        if (du != d[u]) continue;
        for(Data x: g[u]){
            ll v = x.v, uv = x.d, hh = x.h;
            if (hh < LM) continue;
            if (du + uv < d[v]){
                d[v] = d[u] + uv;
                Pre[v] = u;
                pq.push(ii(d[v],v));
            }
        }
    }
}

int main(){
    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    cin >> n >> m >> st >> en;
    for(int i=1; i<=m; i++){
        ll u , v, dd, hh;
        cin >> u >> v >> dd >> hh;
        g[u].push_back(Data(v,dd,hh));
        g[v].push_back(Data(u,dd,hh));
    }

    ll l = 0, r= 1e16;
    while (l<=r){
        ll mid = (l+r)/2;
        dem++;
        ok = false;
        DFS(st, mid);
        if (ok) {
            LM = mid;
            l = mid+1;
        }
        else r = mid-1;
    }
    Dijkstra();
    cout << LM << endl;
    Trace(en);
    return 0;
}
