Thuật toán Z - Lý thuyết

2020-10-17
Lý thuyết So khớp chuỗi , Chuỗi

cp handbook

#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
 
string s;
std::vector<int> z;
 
int main(int argc, char const *argv[])
{
	ios_base::sync_with_stdio(0);
	cin.tie(0); cout.tie(0);
	
	#ifndef ONLINE_JUDGE
	    // freopen("input.txt","r",stdin);
	#endif
	
	
	cin >> s;
	int n = s.size();
	z.resize(n,0);
 
	z[0] = n;
	int l =0, r= 0 ;
	for(int i=1; i<n; i++){
		int x = max(0, min(r-i+1, z[i-l]));
		while (i+x < n && s[x] == s[i+x]) x++;
		z[i] = x;
		if (x + i -1> r){
			l = i;
			r = i + x -1;
		}
	}
	for(int i=1; i< n; i++){
		if (z[i] + i == n) cout << i << " ";
	}
	cout << n;
	return 0;
}
