The Fair Nut and Rectangles - Codeforces

2020-10-03
Quy hoạch động bao lồi Quy hoạch động , Codeforces

Link đề bài:

  • https://codeforces.com/contest/1083/problem/E/.

Hướng dẫn:

  • Đầu tiên ta sắp xếp các hình chữ nhật theo chiều dài tăng dần, mà đề nói không có hình chữ nhật nào chứa hình khác \(->\) Chiều rộng các hình chữ nhật sau khi sắp xếp sẽ giảm dần.
  • Gọi f[i] là cách chọn tối ưu để chọn một số hình chữ nhật từ 1 tới i:
    • Ta có: f[i] = max(f[j] + h[i]*w[i] - h[i]*w[j] - a[i]) \((0 <j < i)\) (hcn thứ \(j\) là hình cuối cùng ta chọn trước khi chọn hcn thứ \(i\)).
  • Để giảm độ phức tạp của bài toán ta có thể dùng qhd bao lồi.

Code mẫu:

#include <bits/stdc++.h>
#define pb push_back
#define FOR(i, a, b) for(int i = int(a); i <= int(b); i++)
#define FORD(i, a, b) for(int i = int(a); i >= int(b); i--)
#define REP(i, r) for(int i = 0; i < r; i++)
#define FAST ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
#define READFILE freopen("input.txt", "r", stdin)
#define WRITEFILE freopen("output.txt", "w", stdout);
#define ll long long
#define ii pair<ll, ll>
#define fi first
#define se second
using namespace std;

const int MAX=-1;

struct Rect
{
	ll x, y, cost;
	Rect(ll _x=0, ll _y=0, ll _cost =0) {x= _x; y =_y; cost = _cost;}
};

bool operator < (Rect a, Rect b){
	if (a.x == b.x) return a.y < b.y;
	return a.x < b.x;
}

std::vector<Rect> a;
std::vector<ll> A;
std::vector<ll> B;

int n, pt;

bool Bad(int l1, int l2, int l3){
	return (1ll*B[l3]-B[l1])*(A[l1]-A[l2])>1ll*(B[l2]-B[l1])*(A[l1]-A[l3]);
}

void Add(ll a, ll b){
	A.push_back(a);
	B.push_back(b);
	while (A.size() >=3 && Bad(A.size()-3, A.size()-2, A.size()-1)){
		A.erase(A.end()-2);
		B.erase(B.end()-2);
	}
}

ll Query(ll x){
	if (pt >= A.size()) pt = A.size() -1;
	while (pt < A.size()-1 && A[pt]*x + B[pt] < A[pt+1]*x + B[pt+1]) pt++;
	return A[pt]*x + B[pt];
}

int main(int argc, char const *argv[])
{
	FAST;
	// freopen("input.txt","r",stdin);
	cin >> n;
	a.resize(n+1);
	for(int i=1; i<=n; i++){
		cin >> a[i].x >> a[i].y >> a[i].cost;
	}
	sort(a.begin()+1, a.begin()+n+1);// x tang -> y giam 
	Add(0,0);
	ll Ans = 0;
	for(int i=1; i<=n; i++){
		ll tmp = Query(a[i].y) - a[i].cost ;
		tmp+=a[i].x * a[i].y;
		Ans = max(Ans, tmp);
		Add(-a[i].x, tmp);
	}
	cout << Ans;
	return 0;
}
