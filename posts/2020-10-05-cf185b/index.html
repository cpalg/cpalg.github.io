Cats Transport - Codeforces

#include <iostream>
#include <vector>
#include <algorithm>
#define ll long long
#define ii pair<ll, ll>
#define fi first
#define se second
using namespace std;

const int MAX = 1e5+6;

std::vector<ll> A;
std::vector<ll> B;
int n, m, k, pt;
ll a[MAX], c[MAX], f[MAX][123], s[MAX];
ii b[MAX];

bool Bad(int l1, int l2, int l3){
	return (B[l3] - B[l1]) * (A[l1] - A[l2]) < (B[l2] - B[l1]) * (A[l1] - A[l3]);
}

void Add(ll a, ll b){
	A.push_back(a);
	B.push_back(b);
	while (A.size()>=3 && Bad(A.size()-3, A.size()-2, A.size()-1)){
		A.erase(A.end()-2);
		B.erase(B.end()-2);
	}
}

ll Query(ll x){
	int l =0, r = A.size()-1;
	while (l<r){
		int mid = (l+r)/2;
		ll f1 = A[mid]*x + B[mid];
		ll f2 = A[mid+1]*x + B[mid+1];
		if (f1 <= f2){
			l = mid+1;
		}
		else r = mid;
	}
	return A[l]*x + B[l];
}

int main(int argc, char const *argv[])
{
	ios_base::sync_with_stdio(0);
	cin.tie(0); cout.tie(0);
	
	#ifndef ONLINE_JUDGE
	    freopen("input.txt","r",stdin);
	#endif
	
	cin >> n >> m >> k;
	for(int i=2; i<=n; i++){
		cin >> a[i];
		a[i] += a[i-1];
	}
	for(int i=1; i<=m; i++){
		cin >> b[i].fi >> b[i].se;
	}
	for(int i=1; i<=m; i++){
		c[i] = b[i].se - a[b[i].fi];
	}
	sort(c+1, c+m+1); 
	for (int i = 1; i <= m; i++) c[i] = c[m] - c[i];
	reverse(c+1, c + m +1);

	for(int i=1; i<=m; i++) s[i] = s[i-1] + c[i];

	for(int i=1; i<=m; i++) f[i][1] = s[i];

	for(int j=2; j<=k; j++){
		A.clear(); B.clear(); pt = 0;
		for(int i=1; i<=m; i++){
			Add(c[i], s[i-1] - f[i-1][j-1] - c[i]*(i-1));
			if (i > 1) f[i][j] = s[i] - Query(i); 
			else f[i][j] = 0;
		}
	}
	cout << f[m][k];
	return 0;
}
