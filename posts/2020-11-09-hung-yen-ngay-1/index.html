Tìm kiếm tam phân.

#include <bits/stdc++.h>
#define ii vector<int, int>
#define fi first
#define se second
using namespace std;

const int MAX = 3e5+6;

long long a[MAX], b[MAX], n;

long long Cal(long long x){
	long long ret = abs(x - a[n/2+1]) + abs(x - b[n/2+1]);
	int cnt = 0;
	for(int i=n/2; i>0; i--){
		cnt++;
		ret+= abs(a[i] - x - cnt) + abs(b[i] - x - cnt);
	}
	cnt = 0;
	for(int i=n/2+2; i<n+1; i++){
		cnt++;
		ret+= abs(a[i] - x - cnt) + abs(b[i] - x - cnt);
	}
	return ret;
}

int main(int argc, char const *argv[])
{
	ios_base::sync_with_stdio(0);
	cin.tie(0); cout.tie(0);
	
	#ifndef ONLINE_JUDGE
	    freopen("BRICKS.INP","r", stdin);
	    freopen("BRICKS.OUT","w", stdout);

		// freopen("input.txt","r", stdin);
	#endif
	
	cin >> n;
	for(int i=1; i<=n; i++){
		cin >> a[i];
	}
	for(int i=1; i<=n; i++){
		cin >> b[i];
	}
	long long l = 1, r = 1e12;

	while (l+2<r){
		long long m1 = (l*2+r)/3, m2 = (l+r*2)/3;
		if (Cal(m1) < Cal(m2)){
			r = m2;
		} 
		else l = m1; 
	}
	cout << min(min(Cal(l), Cal(l+2)), min(Cal(r), Cal(l+1)));
	return 0;
}

Bài 2:

  • Ta có công thức quy hoạch động: f[i] = f[j] + d(i,j) với i là tuyến xe buýt đang xét, j thuộc tập các tuyến xe buýt mà j.u < i.x và j.v < i.y (Theo đề bài không được đi lùi).
  • Để giảm độ phức tạp xuống nlogn thì ta tách một tuyến xe buýt thành 2 điểm, một điểm đầu và một điểm đích. Sắp xếp các tuyến xe buýt theo x tăng dần.
  • Để chiều y tăng dần thì ta nén số và lưu vào cây BIT.
  • Tới mỗi điểm mở đầu thì ta sẽ tính f[], còn điểm kết thúc thì ta sẽ cập nhật.
  • Để tiện thì tạo thêm điểm (n, n), là điểm dừng chân đỡ phải tính lại. Kết quả là f[n][n].

const int MAX = 3e5+6;

int BIT[MAX], f[MAX], n, m;
iiii a[MAX];
std::vector<int> d;
 
int Find(int x){
	return upper_bound(d.begin(), d.end(), x) - d.begin() + 2;
}

ll getBit(int x){
	int64_t ret = 0;
	for(; x>0; x-=x&-x) ret= min(BIT[x], ret);
	return ret;
}

void Update(int x, int t){
	for(; x<MAX; x+=x&-x) BIT[x]= min(BIT[x], t);
}

Bài 3:

  • Cách 1: dùng euler tour, fenwick tree update đoạn.
  • Cách 2:
    • Sắp xếp các điểm theo thứ tự duyệt DFS.
    • Có 2 trường hợp.
      • nếu đỉnh đó nhận một đỉnh đã từng duyệt trước là tổ tiên thì không cần xét.
      • Nếu đỉnh đó không nhận một đỉnh trước là tổ tiên thì ta cộng kết quả thêm số con của đỉnh đó, đồng thời lưu lại đỉnh đó, sẽ là tổ tiên của các đỉnh tiếp theo.
#include <bits/stdc++.h>
#define ii vector<int, int>
#define fi first
#define se second
using namespace std;

const int MAX = 2e5+6;

int cnt, In[MAX], Out[MAX], nChild[MAX], n , m, k;
std::vector<int> g[MAX];

void DFS(int u, int par){
	In[u] = ++cnt;
	nChild[u] = 1;
	for(int i=0; i< (int)g[u].size(); i++){
		int v = g[u][i];
		if (v==par) continue;
		DFS(v, u);
		nChild[u] += nChild[v];
	}
	Out[u] = ++cnt;
}

bool Child(int u, int v){
	return In[u] < In[v] && Out[u] > Out[v];
}

bool cmp(int a, int b){
	return In[a] < In[b];
}

int main(int argc, char const *argv[])
{
	ios_base::sync_with_stdio(0);
	cin.tie(0); cout.tie(0);
	
	#ifndef ONLINE_JUDGE
	    // freopen("input.txt","r",stdin);
		freopen("MAILGROUP.INP", "r", stdin);
		freopen("MAILGROUP.OUT", "w", stdout);
	#endif
	
	cin >> n >> m;
	for(int i=2; i<=n; i++){
		int tmp;
		cin >> tmp;
		g[i].push_back(tmp);
		g[tmp].push_back(i);
	}	
	DFS(1,1);

	for(int i=1; i<=m; i++){
		vector<int> a;
		int res = 0;
		int last = 0;
		In[last] = 0; Out[last] = 0;
		cin >> k;
		for(int i=1; i<=k; i++){
			int tmp;
			cin >> tmp;
			a.push_back(tmp);
		}
		sort(a.begin(), a.end(), cmp);
		for(int i=0; i<(int)a.size(); i++){
			if (!Child(last, a[i])){
				last = a[i];
				res += nChild[a[i]];
			}
		}
		cout << res << "\n";
	}
	return 0;
}
