Edge Directions - CSES

2020-10-17
CSES DFS

Link đề bài:

  • https://cses.fi/problemset/task/1756/.

Hướng dẫn:

Code mẫu:

#include <iostream>
#include <algorithm>
#include <vector>
#define ii pair<int, int>
#define fi first
#define se second
using namespace std;

const int MAX = 2e5+6;

int n, m, vs[MAX], t[MAX];
std::vector<int> g[MAX];
std::vector<ii> res;

void DFS(int u, int par = 0){
	vs[u] = 1;
	for(int v: g[u]){
		if (v == par) continue;
		if (vs[v] == 1) continue;
		if (!vs[v]){
			res.push_back(ii(u, v));
			DFS(v, u);
		} 
		else{
			res.push_back(ii(u ,v));
		}
	}
	vs[u] = 2;
}

int main(int argc, char const *argv[])
{
	ios_base::sync_with_stdio(0);
	cin.tie(0); cout.tie(0);
	
	#ifndef ONLINE_JUDGE
	    freopen("input.txt","r",stdin);
	#endif
	
	
	cin >> n >> m;
	for(int i=1; i<=m; i++){
		int u , v;
		cin >> u >> v;
		g[u].push_back(v);
		g[v].push_back(u);
	}
	for(int i=1; i<=n; i++){
		if (!vs[i]) DFS(i);
	}

	for(int i=0; i< (int)res.size(); i++){
		cout << res[i].fi << " " << res[i].se << "\n";
	}
	return 0;
}
